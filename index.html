							@property
							def offspring(self):
								stack = [self]
								leaves = []
								nodes = []
						
								while len(stack) != 0:
									tail = stack.pop()
									if tail.left or tail.right:
										nodes.append(tail)
										if tail.left: stack.append(tail.left)
										if tail.right: stack.append(tail.right)
									else:
										leaves.append(tail)
						
								return leaves, nodes
